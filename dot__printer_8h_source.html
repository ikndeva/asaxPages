29 virtual void print_dot(Sindex<NT> *G, string outfile) override {
30 if (this->lv.test(Level::TRACE)) cout << "@DotPrinter::print_dot" << endl; assert(G != nullptr);
31 this->G = G;
32
33 //compute the depths of all nodes
34 this->set_top_node(G);
35 this->compute_depth(G, this->top_depth);
36 this->collect_depth(G->get_print_src(), this->H);
37 this->compute_max_depth(this->H);
38
39 //printing dot file
40 print_dot_main(G, outfile);
41 }

45 void compute_depth(Sindex<NT> *G, int top_depth_) override {
46 if (this->lv.test(Level::TRACE)) cout << "@DotPrinterImpl::compute_depth" << endl;
47 compute_depth_rec(G->get_print_src(), top_depth_);
48 }

51 // * ここに、各頂点の深さは根からその頂点までの最長パスの長さ（枝数）で定義する。
52 // * 根の深さは0である。
53 void compute_depth_rec(NT *v, int depth) {
54 if (v == nullptr) return;
55 else if (depth <= v->depth) return;
56
57 if (this->lv.test(Level::TRACE2)) cout << "debug: node=" << v->id << ": depth=" << v->depth << "=> new depth=" << depth << endl;
58
59 v->depth = depth; //update
60 for (auto it : v->childlist()) {
61 NT *w = it.second;
62 compute_depth_rec(w, depth+1);
63 }
64 }

66 virtual void print_dot_main(Sindex<NT> *G, string outfile) {
67 if (this->lv.test(Level::TRACE)) cout << "@DotPrinter::print_dot_main" << endl;
69 ofstream outs(outfile);
70
71 //=====
72 //write the header part
73 outs << this->dot_header_all("G1") << endl;
74 outs << this->dot_header_graph() << endl;
75 outs << this->dot_header_node() << endl;
76 outs << this->dot_header_edge() << endl;
77
78 //=====
79 std::vector<DotLabeledEdge<NT> *> stemList;
80 std::vector<DotLabeledEdge<NT> *> sideList;
81 std::vector<NT *> suffNodeList;
82
83 //print the stem edges between primary nodes
84 for (NT *v0 : G->get_node_factory()->node_list()) {
85 NT *v = (NT *)v0;
86 outs << node_to_dot(v) << endl; //write a node
87
88 for (auto it : v->childlist()) {
89 int ch = it.first;
90 NT *w = it.second;
91 if (this->lv.test(Level::TRACE2))
92 cout << "edge with primality"
93 << Visitable<NT>::edge_to_str(v, w)
94 << ":" << Visitable<NT>::node_to_str(v) << "->is_heavy=" << v->is_heavy
95 << ":" << Visitable<NT>::node_to_str(w) << "->is_heavy=" << w->is_heavy
96 <<" :ep=" << v->get_primary(ch)
97 << endl;
98
102 if (v->is_heavy && w->is_heavy && v->get_primary(ch))
103 stemList.push_back(new DotLabeledEdge<NT>(v, ch, w));
104 else
105 sideList.push_back(new DotLabeledEdge<NT>(v, ch, w));
106
107
108 }
109 suffNodeList.push_back(v);
110 }//for node
111
112 //=====
113 //print a subgraph: the stem edges
114 StylePen *pen_stem = new StylePen(); //heavy line
115 pen_stem->weight = StylePen::WEIGHT_HEAVY;
116 pen_stem->dir = StylePen::DIR_FORWARD;
117 outs << endl << "subgraph stemedges {" << endl;
118 for (auto it : stemList) {
119 pen_stem->width = (it->src->get_primary(it->dst->label_ch) ?
120 StylePen::WIDTH_THICK : StylePen::WIDTH_NORMAL);
121 outs << this->edge_to_dot(it->src, it->dst, pen_stem) << endl; //write an edge
122 }
123 outs << "}" << endl;
124
125 //=====
126 //print a subgraph: the side edges
127 outs << endl << "subgraph sideedges {" << endl;
128 StylePen *pen_side = new StylePen(); //heavy line
129 pen_side->weight = StylePen::WEIGHT_NORMAL;
130 pen_side->dir = StylePen::DIR_FORWARD;
131 for (auto it : sideList) {
132 pen_side->width = (it->src->get_primary(it->dst->label_ch) ?
133 StylePen::WIDTH_THICK : StylePen::WIDTH_NORMAL);
134 outs << this->edge_to_dot(it->src, it->dst, pen_side) << endl; //write an edge
135 }
136 outs << "}" << endl;
137
138 //=====
139 //print a subgraph: the suf edges (slinks)
140 StylePen *pen_suf = new StylePen();
141 pen_suf->weight = StylePen::WEIGHT_NORMAL;
142 pen_suf->dir = StylePen::DIR_BACKWARD;
143 pen_suf->color = StylePen::COLOR_RED;
144 pen_suf->fontcolor = StylePen::COLOR_RED;
145 outs << endl << "subgraph suflinks {" << endl;
146 for (auto it : suffNodeList) {
147 outs << this->suf_to_dot(it, pen_suf) << endl; //write a suflink
148 }
149 outs << "}" << endl;
150
151 //=====
152 //align nodes in the same level
153 this->rank_to_dot(G, outs, this->top_node, this->maxdepth);
154
155 //=====
156 //write the trailer part
157 outs << this->dot_trailer_all() << endl;
158
159 }//print_dot

163 std::string node_to_dot(NT* v) override {
164 assert(v != nullptr);
165 this->ss.str(""); //clear
166
167 //special case for a master node
168 std::string str_prop_master = "";

172 //body

183 // * char型keyに対して、構築子std::string(1,key)は長さ1指定の文字列を生成。
184 virtual std::string edge_to_dot(NT* v, NT *w, StylePen *pen) override {
185 assert(this->G != nullptr);
186 assert(v != nullptr);
187 assert(w != nullptr);
188
189 //for debug
190 std::string str_debug_ =
191 this->str_prop("_primary", v->get_primary(w->label_ch))+
192 this->str_prop("_heavy_src", v->is_heavy)+
193 this->str_prop("_heavy_dst", w->is_heavy)+
194 "";
195
196 std::string lab = this->G->get_label_string(v, w);
197 if (this->lv.test(Level::TRACE2))
198 cout << "debug: lab=[" << this->G->get_label_string(v, w) << "]" << endl;
199 this->ss.str(""); //clear
200 this->ss << this->str_edge(v, w)
201 << this->str_style(""+
202 this->str_prop("label", lab)+
203 this->str_prop("color", pen->color)+
204 this->str_prop("style", pen->style)+
205 this->str_prop("penwidth", pen->width)+
206 this->str_prop("weight", pen->weight)+
207 this->str_prop("dir", pen->dir)+
208 this->str_prop("fontcolor", pen->fontcolor)+
209 str_debug_+
210 "");
211 return this->ss.str();
212 }
