116 void collect_depth(NT *v, multimap<int, NT *> &H) {
117 if (v == nullptr) return;
118 for (auto it : v->childlist()) {
119 NT *w = it.second;
120 collect_depth(w, H);
121 }
122 H.insert(std::make_pair(v->depth, v));
123 }
124
125 void compute_max_depth(multimap<int,NT *> &H) {
126 int maxd_ = this->top_depth;
127 for (auto it : H) {
128 if (it.first > maxd_) maxd_ = it.first; //it.first: depth
129 }
130 this->maxdepth = maxd_;
131 }

135 //the top node
136 void rank_to_dot(Sindex<NT> *G, ofstream &outs, NT *top_node, int maxdepth) {
137 //print the rank of top node
138 outs << "{ rank = min; " << top_node->id << "}" << endl;
139
140 //print the ranks of the rest
141 for (int d = 0; d <= maxdepth; d++) {
142 outs << "{ rank = same; ";
143 for (NT *v : G->get_node_factory()->node_list()) {
144 if (v->depth == d)
145 outs << "" << v->id << "; ";
146 }
147 outs << "}" << endl;
148 }
