81 void insert(string str) override {
82 if (lv.test(Level::VERBOSE)) cout << "insert_suffices: \n";
83 int ch; //an input letter
84 MapEdgeNode *head = nullptr; //the locus of the present longest inserted suffix
85 MapEdgeNode *curr = nullptr; //the active pointer for insertion
86 MapEdgeNode *child = nullptr; //the ch-child of curr
87
88 MapEdgeNode *sink = nullptr;
89 MapEdgeNode *newsink = nullptr;
90 MapEdgeNode *last = nullptr;
91
92 sink = this->src; //the root
93 //loop for each letter
94 for (int j = 0; j < str.size(); j++) {
95 ch = str.at(j);
96 this->dset->push_back((char)ch); //slab:new
97
98 //add a newsink
99 curr = sink;
100 if (lv.test(Level::TRACE)) cout << "!== insert ch='" << (char)ch << "' starting at node"<<curr->id <<" ==\n";
101
102 //traverse suffix links
103 newsink = nullptr; //note: nullptr stands for "not yet used"
104 while (true) {
105
106 if ((child = (MapEdgeNode *)curr->get(ch)) != nullptr)
107 break; //Type-III: exit the whie-loop
108
109 //Type-II: add a new leaf child
110 if (newsink == nullptr) { //first time in a border
111 newsink = this->NF->get_node();
112 newsink->is_heavy=true;
113 curr->put_primary(ch);
114 newsink->ldep = curr->ldep + 1; //SLAB
115 }
116 curr->put(ch, newsink);
117
118 if (lv.test(Level::TRACE))
119 cout << "@type2: "
120 << curr->edge_to_str(newsink)
121 << " by ch=" << (char)ch
122 << " prim=" << curr->get_primary(ch) << endl;
123
124 curr = curr->get_suf();
125 } //while
126
127 if (lv.test(Level::TRACE)) cout << "@type3: node" << curr->id << " => node" << child->id << " by ch=" << (char)ch << " prim=" << curr->get_primary(ch) << "\n";
128
129 //Assumption: child = curr->get(ch) surely exists.
130 assert(child != nullptr);
131 // On a primary edge: Do nothing
132 if (curr->get_primary(ch)) {
133 if (newsink != nullptr) {
134 newsink->suf = child; //set slink
135 suf_manage_send(newsink); //slab
136 }
137 }
138 else { //On a secondary edge: Split
139 assert(child != nullptr); //for debug
140 assert(curr != nullptr); //for debug
141
142 //create a newchild as a copy of child
143 MapEdgeNode *newchild = clone(child);
144 curr->put(ch, newchild);
145 curr->put_primary(ch);
146 newchild->ldep = curr->ldep + 1; //<==SLAB
147
148 //split point
149 int smid = child->sbeg + (child->ldep - newchild->ldep);
150 MapEdgeNode *childsuf = child->suf;
151
152 repo_node(child, "child");
153 repo_node(newchild, "newchild");
154 repo_node(childsuf, "childsuf");
155
156 newchild->suf = childsuf;
157 newchild->sbeg = smid;
158 newchild->send = child->send;
159
160 child->suf = newchild; //set slink
161 child->sbeg = child->sbeg;
162 child->send = smid;
163
164 if (newsink != nullptr) {
165 newsink->suf = newchild; //set slink
166 suf_manage_send(newsink); //<==SLAB
167 }
168
169 curr = curr->suf;
170 while (curr->get(ch) == child && !curr->get_primary(ch)) {
171 curr->put(ch, newchild);
172 curr = curr->suf;
173 }
174 child = newchild; //experimental: gdawg
175 }
176
177 if (newsink != nullptr)
178 sink = newsink;
179 else
180 sink = child;
181 }//for
182 if (lv.test(Level::TRACE)) cout << "status: exit for-loop\n";
183 }//insert
