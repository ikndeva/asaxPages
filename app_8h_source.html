106 //virtual void readOptions(int argc, char *argv[], Sindex<NT> &G);
107 void readOptions(int argc, char *argv[], Sindex<NT> &G, DotPrintable<NT> &DP) {
108 int C; /* Option letter */
109 extern int optind;
110 extern char *optarg;
111
112 while ((C = getopt(argc, argv, OptionFormat)) != EOFOPT) {
113#ifdef DEBUG_OPTION
114 printf("#getopt: C=%d\n", C);
115#endif
116 switch (C) {
117 //Options for flags
118 case 'g':
119 printAnOption("g", optind, optarg);
120 OPTION_TO_DRAW_GRAPH = true;
121 break;
122
123 case 'n':
124 printAnOption("n", optind, optarg);
125 OPTION_TRACE_LEVEL = 0;
126 // G.TRACE_LEVEL = OPTION_TRACE_LEVEL = 0;
127
128 break;
129
130 case 'v':
131 printAnOption("v", optind, optarg);
132 OPTION_TRACE_LEVEL = 1;
133 //G.TRACE_LEVEL = OPTION_TRACE_LEVEL = 1;
134 break;
135
136 case 'd':
137 printAnOption("d", optind, optarg);
138 DOT_NAM = optarg; //option -d dotfile
139 DOT_FILE = DOT_NAM+"."+DOT_EXT;
140 break;
141
142 case 'f':
143 printAnOption("f", optind, optarg);
144 OPTION_FONT_SIZE = atoi(optarg); //option -f size
145 break;
146
147 case 'i':
148 printAnOption("i", optind, optarg);
149 OPTION_INFILE_NAME = optarg; //option -i infile
150 break;
151
152 case 'z':
153 printAnOption("z", optind, optarg);
154 OPTION_DOC_DELIM = optarg; //option -z end-of-document delimitor
155 break;
156
157 case 't':
158 printAnOption("t", optind, optarg);
159 OPTION_TRACE_LEVEL = atoi(optarg); //option -t level
160 break;
161
162 case 'h':
163 printAnOption("h", optind, optarg);
164 cout << USAGE_STRING << endl;
165 exit(0);
166
167 default:
168 printAnOption("*", optind, optarg);
169 printf("#main: undefined option: -%c\n", C);
170 exit(0);
171 }
172 }
173
174 //post processing: set trace level
175 G.lv.set(OPTION_TRACE_LEVEL);
176 DP.lv.set(OPTION_TRACE_LEVEL);
177 Visitable<NT>::lv.set(OPTION_TRACE_LEVEL);
178
179 if (OPTION_FONT_SIZE > 0) {
180 DP.FONTSIZE_NODE = OPTION_FONT_SIZE;
181 DP.FONTSIZE_EDGE = OPTION_FONT_SIZE;
182 DP.FONTSIZE_SUF = OPTION_FONT_SIZE;
