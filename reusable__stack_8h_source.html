59 ReusableStack(MemoryStoreFactory<S> *factory) {

63 assert(this->_list);

68 ReusableStack(MemoryStoreFactory<S> *factory, int maxlen) : ReusableStack(factory) {
69 this->ensureCapacity(maxlen);

71
72 void clear() {
73 while (!this->empty()) {
74 S *tmp = this->peek();
75 assert(tmp != nullptr);
76 this->_factory->reset(tmp);
77 this->pop();
78 }
79 }

82 void ensureCapacity(int maxlen) {
83 assert(this->_list);
84 while (this->_list->size() < maxlen) {
85 S *tmp = this->_factory->generate();
86 assert(tmp != nullptr);
87 this->_list->push_back(tmp);
88 }
89 assert(this->_list->size() >= maxlen);
90 }
91
93 int capacity() {
94 assert(this->_list);
95 return this->_list->size();
96 }
97
99 int size() {
100 return this->_len;
101 }
102
104 bool empty() { return (this->_len == 0); }
105
107 S *get(int pos) {
108 assert(0 <= pos && pos < this->size());
109 assert(this->_list);
110 S *tmp = this->_list->at(pos);
111 assert(tmp != nullptr);
112 return tmp;
113 }
114
117 bool has_env() { return (this->_len > 0); }
118
121 bool has_last_env() { return (this->_len > 1); }
122
124 S *peek() {
125 assert(!this->empty()); //スタックは少なくとも1つの要素をもつ必要。
126 return this->get(this->_len-1);
127 }
128
130 S *lastpeek() {
131 assert(has_last_env()); //スタックは少なくとも2つの要素をもつ必要。
132 return this->get(this->_len-2);
133 }
134
136 void push() {
137 ensureCapacity(this->_len+1);
138 this->_len++;
139 this->_factory->reset(this->peek());
140 }
141

146 }
