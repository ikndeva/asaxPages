171 if (str == null) IS_SPC_STATE = true;
172 else if (str.length() > 0) IS_SPC_STATE = false;
173 else if (endWithSpace(str)) IS_SPC_STATE = true;
174 else IS_SPC_STATE = false;
175 IS_EMPTY_LINE = false;
176 this.append(str);
177 return this;

181 static boolean endWithSpace(String str) {
182 if (str == null) return true;
183 int len = str.length();
184 if (len == 0) return true;
185 if (str.charAt(len-1) == ' ') return true;
186 else return false;
187 }
