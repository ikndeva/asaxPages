package frame;
import util.*;
import frame.Addable.*;
//public
class StringOutStream implements Addable<String,StringOutStream> {
//======
// Class
//======
public static class StringOutStreamManager extends ContainerManager<StringOutStream> {
StringOutStreamManager() {
this.generator = ()->new StringOutStream();
this.resetter = (StringOutStream holder)->holder.reset();
}
public String toString() { return "StringOutStreamManager"; }
}//class StringOutStreamManager
public static ContainerManager<StringOutStream> manager() { return new StringOutStreamManager(); }
//======
// Methods
//======
static String DEFAULT_DELIM = ", ";
String delim = DEFAULT_DELIM;
private StringBuf buf = null;
private int numStr = 0;
public StringOutStream() {
//this.buf = null;
this.reset();
}
public StringOutStream reset() {
//this.buf.clear();
this.delim = DEFAULT_DELIM;
this.numStr = 0;
return this;
}
public StringOutStream setDelim(String delim) {
this.delim = delim;
return this;
}
public String get() {
Util.errorExit("StringOutStream.get: prohibited to call and not implemented! ");
return null;
}
public StringOutStream add(String str) {
if (this.numStr > 0) System.out.print("["+DEFAULT_DELIM+"]");
System.out.print("["+str+"]");
this.numStr++;
return this;
}
public String toString() { return this.buf.toString(); }
} //class StringOutStream
